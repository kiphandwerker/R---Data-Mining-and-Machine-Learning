70% and 30% Split

n = nrow(X) 
nrep = 20 
mse.train.l = mse.train.r = mse.train.en = rep(0,nrep) 
mse.test.l = mse.test.r = mse.test.en = rep(0,nrep)

set.seed(1)

for(i in 1:nrep) 
{ 
  train_index = sample(1:n,0.7*n,replace = FALSE) # randomly select the indices of the training set 
  X_train = X[train_index,] # trainining X 
  Y_train = Y[train_index] # training Y 
  X_test = X[-c(train_index),] # Test X 
  Y_test = Y[-c(train_index)] # Test Y

  # Ridge 
  cvfitr = cv.glmnet(x=X_train,y=Y_train,family="gaussian",alpha=0) 
  pred.train.r = predict(cvfitr, newx = X_train, s = "lambda.min") 
  pred.test.r = predict(cvfitr, newx = X_test, s = "lambda.min") 
  mse.train.r[i] = mean((Y_train - pred.train.r)^2) 
  mse.test.r[i] = mean((Y_test - pred.test.r)^2)

  #lasso 
  cvfitl = cv.glmnet(x=X_train,y=Y_train,family="gaussian",alpha=1) 
  pred.train.l = predict(cvfitl, newx = X_train, s = "lambda.min") 
  pred.test.l = predict(cvfitl, newx = X_test, s = "lambda.min") 
  mse.train.l[i] = mean((Y_train - pred.train.l)^2) 
  mse.test.l[i] = mean((Y_test - pred.test.l)^2)

  # E - net 
  cvfiten = cv.glmnet(x=X_train,y=Y_train,family="gaussian",alpha=0.5) 
  pred.train.en = predict(cvfiten, newx = X_train, s = "lambda.min") 
  pred.test.en = predict(cvfiten, newx = X_test, s = "lambda.min") 
  mse.train.en[i] = mean((Y_train - pred.train.en)^2) 
  mse.test.en[i] = mean((Y_test - pred.test.en)^2)
}

mse_3 = c(mse.train.r,mse.test.r,mse.train.l,mse.test.l,mse.train.en,mse.test.en)
indicator.3 = as.factor(c(rep("Ridge Training",nrep),
                          rep("Ridge Test",nrep),
                          rep("Lasso Training",nrep),
                          rep("Lasso Test",nrep),
                          rep("ENET Training",nrep),
                          rep("ENET Test",nrep)))

Model Comparison

ggplot(data = as.data.frame(mse_3)) +

                             fill = indicator.3,
                             y = mse_3)) +
  ggtitle("Ridge, Lasso, Elastic Net Comparison ") +
  xlab("Models") +
  ylab("MSE")
