NEURAL NET

train.index = sample(1:nrow(purchase_data),0.5*nrow(purchase_data))
  
std.data = as.data.frame(scale(purchase_data[,-1]))
std.data = cbind("Purchase" = purchase_data$Purchase,std.data)


std.train = std.data[train.index,]
std.test = std.data[-train.index,]

MSE.trnnet = matrix(0,6,1)
MSE.tsnnet = matrix(0,6,1)
for (i in 1:6){
  nnet.fit = nnet(Purchase~., data = std.data, subset = train.index, size = i, rang = 0.1,
                  decay = 5e-4, maxit = 200)
  
  predtr.nnet = predict(nnet.fit,newdata=std.train,type="raw")
  predts.nnet = predict(nnet.fit,newdata=std.test,type="raw")
  std.train.purch = (std.train$Purchase)
  std.test.purch = (std.test$Purchase)
  
  MSE.trnnet[i] = mean((std.train.purch - predtr.nnet)^2)
  MSE.tsnnet[i] = mean((std.test.purch - predts.nnet)^2)
}
MSE.nnet = cbind(MSE.trnnet, MSE.tsnnet)
colnames(MSE.nnet) = c("Train", "Test")
rownames(MSE.nnet) = c("Node 1","Node 2","Node 3","Node 4","Node 5","Node 6")
