Training the Models

set.seed(12345) 
n = nrow(life.expectancy.refined) # the sample size 
training.index = sample(1:n,0.7*n,replace = FALSE) # training set row index 
test.index = setdiff(1:n,training.index) # test set row index 
training.data = life.expectancy.refined[training.index,] # training data 
test.data = life.expectancy.refined[test.index,] # test data 
names(life.expectancy.refined)
##  [1] "LIFEEXP"         "ILLITERATE"      "POP"            
##  [4] "FERTILITY"       "PRIVATEHEALTH"   "PUBLICEDUCATION"
##  [7] "HEALTHEXPEND"    "BIRTHATTEND"     "PHYSICIAN"      
## [10] "GDP"
set.seed(12345) 
prop = c(0.7,0.5) 
train_mse = test_mse = matrix(0,2,2)
for(i in 1:2) 
{
  train_index = sample(1:nrow(life.expectancy.design), 
                       prop[i]*nrow(life.expectancy.design), 
                       replace = FALSE) 
  test_data_ols = data.frame(life.expectancy.design[-train_index,] ) 
  test_data_f = data.frame(life.expectancy.design[-train_index,
                                                  c("FERTILITY","HEALTHEXPEND","BIRTHATTEND")])
  # fitted models 
  fit_ols = lm(LIFEEXP ~ .,data = life.expectancy.refined[train_index,]) 
  fit_f = lm(LIFEEXP ~ FERTILITY + HEALTHEXPEND + BIRTHATTEND,
             data = life.expectancy.refined[train_index,])
  # Obtain Predicted values 
  train_pred_ols = fit_ols$fitted.values
  test_pred_ols = predict(fit_ols,newdata = test_data_ols) 
  train_pred_f = fit_f$fitted.values 
  test_pred_f = predict(fit_f,newdata = test_data_f)

  # Compute MSEs 
  train_mse_ols = mean((life.expectancy.refined[train_index,1]-train_pred_ols)^2) 
  test_mse_ols = mean((life.expectancy.refined[-train_index,1]-test_pred_ols)^2) 
  train_mse_f = mean((life.expectancy.refined[train_index,1]-train_pred_f)^2) 
  test_mse_f = mean((life.expectancy.refined[-train_index,1]-test_pred_f)^2)
  train_mse[i,] = c(train_mse_ols, train_mse_f) 
  test_mse[i,] = c(test_mse_ols, test_mse_f) 
}

colnames(train_mse) = c("OLS_Tr","FSW_Tr") 
rownames(train_mse) = c("70/30","50/50") 
colnames(test_mse) = c("OLS_Ts","FSW_Ts")
rownames(test_mse) = c("70/30","50/50")
