suggest_transformation <- function(x,powers=seq(from=0,to=3,by=0.5),add=0,message=TRUE) {
  if( !(class(x) %in% c("numeric","integer") ) ) { return(NULL) }
  require(robustbase)
  if(add!=0) { x <- x + add }
  if(min(x)<=0) { 
    if(message==TRUE) { 
      cat(paste("Smallest value of x is 0 or smaller...considering 
                only positive powers for transformations\n")) }
    powers <- powers[which(powers>0)] }
  skewnesses <- rep(0,length(powers))
  for (p in powers) {
    if(p==0) { x.trans <- log10(x) } else { x.trans <- x^p }
    skewnesses[which(powers==p)] <- mc(x.trans)
  }
  best.p <- powers[which.min(skewnesses)]
  if(best.p==0) { return("log10")} else { return(best.p) }
  }


scale_dataframe <- function(DATA,except=NA) {
  column.classes <- unlist(lapply(DATA,FUN=class))  
  numeric.columns <- which(column.classes %in% c("numeric","integer"))  
  if(!is.na(except)) { 
    exemptions <- which( names(DATA) %in% except )
    if ( length(exemptions)>0 ) { numeric.columns <- setdiff(numeric.column,exemptions) } }
  if( length(numeric.columns)==0 ) { return(DATA) }
  DATA[,numeric.columns] <- as.data.frame( scale(DATA[,numeric.columns]) )
  return(DATA)
}
